Operating Systems

// Linux commands
ps // list all working processes
pgrep token // return all processes whose name contain token
kill [-signum] pid // send signal signum to process pid
pkill pname // kill process by name

var=value // define BASH variables
echo $var // print variable value
${var} or $var // variable usage
'${var}' // = ${var}, as '' preserves entire content
"${var}" // = var_content, as "" preserves content except for $, ', \, !
${var} // performe word splitting and filename expansion if without quotes
export var // make an environment variable (shell variables are not passed to a program)
$? // get return status of a program if called after it
wait $pid // get return status of pid process

command > file // redirect output stream
command 2> file // redirect error stream
command 2>&1 file // redirect error into output stream
command > file 2>&1 file // redirect error and output into file
command >> file // append to a file (not streams)
command &> file // shortcut for command > file 2>&1 file
command &>> file // shortcut for command >> file 2>&1 file


// General C commands
getenv("var_name") // get environment variable
fflush(stdin) // clear stdin
fflush(stdout) // force all stdout buffer to write


// Process manager
fork() // fork current process and create a child
wait(NULL) // wait for child to finish
wait(&status) // save return value of child into status
WEXITSTATUS(status) // get exit status (0 for ok, 1-127 for errors, 128-... signal number that terminated the process)
WTERMSIG(status) // get signal number that killed the process

getpid() // get current pid
getppid() // get parent pid
waitpid() // wait for specified process

execl() // execute command with list of arguments
execv() // execute command with pointer to array of arguments
execlp(), execvp() // search for PATH environment variable, not absolute path
execlpe(), execvpe() // pass the environment variables to new process


// Signal handling
#include <signal.h>
struct sigaction act
void funct(int signum) {...} // create handler function
act.sa_handler = funct // assign funct() as handler of struct act
sigaction(signum, *act, NULL) // link handler function to signal signum (can be both number and name)


// Threads with OpenMP
#include <omp.h>
// add -fopenmp to gcc arguments
#pragma omp parallel for // parallelize following for
#pragma omp parallel { // parallelize following block
    #pragma omp barrier // force every thread to finish what's before before going on
}
#pragma omp single {} // assign following block to just one thread
#pragma omp parallel sections { // define sections to be executed by one thread each
    #pragma omp section {}
    #pragma omp section {}
}
#pragma omp critical {} // define critical section which blocks all other threads
#pragma omp atomic // following update is forced to be atomic
#pragma omp [...] shared(var) // set shared variable
#pragma omp [...] private(var) // set private variable for each thread
// OMP_NUM_THREADS=4 ./a.out // set number of threads (env variable)

// Threads with pthreads
#include <pthread.h>
// add -lpthread to gcc arguments
pthread_t tid // create thread id
pthread_attr_t attr
pthread_attr_init(*attr) // set thread attributes
pthread_create(*tid, *attr, *funct, *args) // create new thread and assign the function to it, passing arguments
pthread_join(tid, **retval) // wait tid thread to finish and get its return value into retval
pthread_exit(*retval) // terminate calling thread
pthread_kill(tid, signum) // send signal to thread
pthread_cancel(tid) // send cancel signal to thread


// Atomic variables
#include <stdatomic.h> 
atomic_read(*var) // read variable
atomic_store(*var, value) // set var = value
atomic_fetch_add(*var, value) // set var += value while returning old value of var
atomic_fetch_sub(*var, value) // set var -= value while returning old value of var
atomic_flag_test_and_set(*lock) // wait until lock is acquired
atomic_flag_clear(*lock) // release lock


// POSIX API mutex
pthread_mutex_t mutex
pthread_mutex_init(&mutex, NULL) // initialize mutex
pthread_mutex_lock(&mutex) // lock mutex
pthread_mutex_unlock(&mutex) // unlock mutex
pthread_mutex_destroy(&mutex) // delete mutex

// Semaphores
sem_t *sem = sem_open("/name", O_CREAT, 0644, 1) // create named (global) semaphore (name always starts with '/')
sem_wait(*sem)
sem_post(*sem) // increase semaphore
sem_close(*sem) // close semaphore for calling thread
sem_unlink(*sem) // delete named semaphore for all processes

sem_t sem; // create unnamed semaphore (accessible just for threads in a process)
sem_init(*sem, 0, val); // initialize unnamed semaphore (second argument) with value val
sem_destroy(*sem) // delete unnamed semaphore

// Conditional variables
pthread_cond_t cvar; // create conditional variable
pthread_cond_init(*cvar, NULL); // init conditional variable


// I/O operations
open(path, mode [| flags]) // open file in <O_RDONLY, O_WRONLY, O_RDWR> modes. Flags are inserted like O_WRONLY | O_APPEND
int fd = open() // collect file descriptor
write(fd, buffer, sizeof(buffer)) // write buffer into fd file. Sizeof can be replaced with number of bits to write
read(fd, buffer, sizeof(buffer)) // read fd into buffer. Sizeof can be replaced with number of bits to read
close(fd) // close file
